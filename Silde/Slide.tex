\section{2. Mục tiêu hỗ trợ ra quyết định}
% ==========================================

\begin{frame}{2.1 Quyết định phân bổ ngân sách khuyến mãi}
    \textbf{Bài toán:} Có nên cấp phát mã giảm giá (Voucher) cho khách hàng trong phiên hiện tại không? Mức giảm là bao nhiêu?
    
    \vspace{0.3cm}
    \textbf{Giải pháp từ mô hình AI:} Thay vì "rải" mã giảm giá đại trà, hệ thống chia khách hàng làm 3 nhóm:
    \begin{itemize}
        \item \textbf{Chắc chắn mua (Xác suất $>$ 80\%):} Khách đã có ý định rõ ràng. $\rightarrow$ \textit{Không tặng Voucher} để bảo vệ biên lợi nhuận.
        \item \textbf{Đang phân vân (Xác suất 40\% - 80\%):} Khách thêm đồ vào giỏ nhưng chần chừ. $\rightarrow$ \textit{Tặng Pop-up Voucher 10 phút} để ép hiệu ứng FOMO.
        \item \textbf{Không có ý định (Xác suất $<$ 40\%):} Chỉ lướt xem giá. $\rightarrow$ \textit{Không tặng Voucher} để tránh lãng phí.
    \end{itemize}
\end{frame}

\begin{frame}{2.2 Quyết định quảng cáo bám đuôi \& Can thiệp CSKH}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \textbf{Tối ưu Quảng cáo Retargeting:}
            \begin{itemize}
                \item Chạy quảng cáo cho 100\% người thoát trang là "đốt tiền".
                \item \textbf{Quyết định:} Chỉ lọc Top 20\% khách hàng có xác suất mua cao nhất đẩy lên Facebook/Google. 
                \item $\rightarrow$ Tiết kiệm CAC, tăng vọt ROAS.
            \end{itemize}
        \end{column}
        
        \begin{column}{0.5\textwidth}
            \textbf{Can thiệp thời gian thực (CSKH):}
            \begin{itemize}
                \item Khách có xác suất chốt đơn cao nhưng dừng ở trang thanh toán quá 3 phút?
                \item \textbf{Quyết định:} Hệ thống báo động để nhân viên chủ động Live-chat: \textit{"Chào bạn, bạn có cần hỗ trợ phí vận chuyển không?"}
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{5.1 Chiến lược Đánh giá \& Xử lý Mất cân bằng}
    \textbf{Mục tiêu Đánh giá:}
    \begin{itemize}
        \item Bỏ qua Accuracy. Tối ưu tuyệt đối vào chỉ số \textbf{ROC-AUC}.
        \item Theo dõi sát \textbf{Recall} để đảm bảo không bỏ lọt khách hàng.
    \end{itemize}

    \vspace{0.4cm}
    \textbf{Cấu hình Trọng số chống thiên lệch (Class Weights):} \\
    Ép thuật toán "chú ý" vào nhóm chốt đơn thông qua công thức:
    $$ W_j = \frac{N_{total}}{k \times N_j} $$
    \begin{itemize}
        \item \textbf{Trọng số Nhãn 0 ($W_0$):} $0.516$
        \item \textbf{Trọng số Nhãn 1 ($W_1$):} $16.110$
        \item \textit{Phạt nặng gấp 31 lần nếu đoán sai khách hàng có mua hàng!}
    \end{itemize}
\end{frame}

\begin{frame}{7.1 Khuyến nghị Ra quyết định (Marketing \& Khuyến mãi)}
    \textbf{1. Chuyển dịch sang Marketing mục tiêu:}
    \begin{itemize}
        \item Ngừng chạy Ads bám đuôi đại trà. 
        \item Tập trung ngân sách vào tệp được AI gán \textbf{Nhãn 1}.
        \item \textit{Hiệu quả:} Giảm mạnh CAC, tăng ROAS gấp nhiều lần do tệp khách hàng được tinh lọc.
    \end{itemize}

    \vspace{0.4cm}
    \textbf{2. Phân phối Khuyến mãi Thông minh:}
    \begin{itemize}
        \item Khách chắc chắn mua (Xác suất $>90\%$): \textbf{Không cấp mã}.
        \item Khách phân vân (Xác suất $70-90\%$): \textbf{Bật Pop-up Voucher 15 phút}.
        \item Khách lướt dạo (Xác suất $<50\%$): \textbf{Gợi ý đọc Blog} để lấy Email.
    \end{itemize}
\end{frame}

\begin{frame}{7.2 Khuyến nghị Ra quyết định (CSKH \& Giao diện)}
    \textbf{3. Can thiệp Chăm sóc Khách hàng Thời gian thực:}
    \begin{itemize}
        \item Tích hợp mô hình vào Backend qua Apache Kafka.
        \item Nếu AI phát hiện phiên có \textit{Xác suất cao + Dừng quá lâu ở Checkout}.
        \item $\rightarrow$ Báo động cho CSKH vào Live-chat can thiệp ngay để cứu đơn hàng.
    \end{itemize}

    \vspace{0.4cm}
    \textbf{4. Tối ưu Trải nghiệm (Dynamic UI/UX):}
    \begin{itemize}
        \item \textbf{Nhóm 0:} Giao diện Discovery (Khám phá) nhiều hình ảnh.
        \item \textbf{Nhóm 1:} Giao diện Checkout. Ẩn quảng cáo rườm rà, đẩy nút \textbf{Thanh toán} ra giữa màn hình.
    \end{itemize}
\end{frame}

\begin{frame}{7.3 Đánh giá Rủi ro và Hạn chế}
    \textbf{Rủi ro Triển khai Kinh doanh:}
    \begin{itemize}
        \item \textbf{Thất thoát do False Positives:} Precision 24\% nghĩa là trong 100 người dự đoán mua, có 76 người không mua. \textbf{Tuyệt đối không} dùng mô hình này để tặng Voucher tiền mặt lớn hay gọi điện Telesales. Chỉ hợp với Email/Push Notification.
        \item \textbf{Thao túng AI:} Khách hàng có thể cố tình giả vờ để ép AI nhả mã giảm giá. Cần giới hạn Rate Limiting.
        \item \textbf{Bắt buộc:} Phải chạy \textbf{A/B Testing} trước khi đưa ra toàn hệ thống.
    \end{itemize}

    \vspace{0.3cm}
    \textbf{Hạn chế của Kỹ thuật hiện tại:}
    \begin{itemize}
        \item Chỉ dự đoán trong 1 phiên duy nhất, chưa nối được chuỗi hành trình vắt chéo thiết bị (Cross-device).
        \item Dữ liệu dạng đếm (count), đánh mất tính tuần tự (Time-series) của click.
    \end{itemize}
\end{frame}

\begin{frame}{7.4 Hướng phát triển trong tương lai}
    \textbf{Định hướng nâng cấp Hệ thống AI:}
    \begin{itemize}
        \item \textbf{Ứng dụng Học Sâu (Deep Learning):} Chuyển sang mô hình Dữ liệu chuỗi. Sử dụng \textbf{RNN, LSTM} hoặc \textbf{Transformer} để AI "hiểu" được tuần tự click chuột (Ví dụ: View $\rightarrow$ Cart sẽ khác với Cart $\rightarrow$ Remove).
        \item \textbf{Hệ thống MLOps:} 
        \begin{itemize}
            \item Hứng dữ liệu Clickstream theo thời gian thực (Spark Streaming).
            \item Huấn luyện liên tục (Continuous Training) để chống hiện tượng trôi dạt dữ liệu (Data Drift) do mùa vụ.
        \end{itemize}
    \end{itemize}
\end{frame}
